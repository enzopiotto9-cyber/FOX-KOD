      function escapeHtml(str){
        return String(str ?? '').replace(/[&<>"']/g, c=>({
          '&':'&amp;', '<':'<', '>':'>', '"':'"', "'":'&#039;'
        }[c]));
      }

      function pad2(n){return String(n).padStart(2,'0')}

      function formatHMS(ms){
        const total = Math.max(0, Math.floor(ms/1000));
        const h = Math.floor(total/3600);
        const m = Math.floor((total%3600)/60);
        const s = total%60;
        return `${pad2(h)}:${pad2(m)}:${pad2(s)}`;
      }
      function parseHMS(str){
        const m = String(str||'').trim().match(/^(\d{1,2}):(\d{1,2}):(\d{1,2})$/);
        if(!m) return null;
        const h = Number(m[1]);
        const mi = Number(m[2]);
        const s = Number(m[3]);
        if(mi>59 || s>59) return null;
        return (h*3600 + mi*60 + s)*1000;
      }
      function safeNumber(x){
        const n = Number(x);
        return Number.isFinite(n) ? n : null;
      }

      function evaluateExpression(expr){
        if(!/^[0-9+\-*/.\s]+$/.test(expr)) return null;
        try{
          const res = Function('"use strict"; return ('+expr+');')();
          if(!Number.isFinite(res)) return null;

      function appendToken(tok){
        if(tok==='.'){
          const parts = calcState.expr.split(/([+\-*/])/).filter(Boolean);
          const last = parts[parts.length-1] ?? '';
          if(last.includes('.')) return;
        }
        calcState.expr += tok;
      }
      function onKey(tok){
        if(tok==='C'){
          calcState.expr='';
          calcState.value='0';
          updateCalcUI();
          return;
        }
        if(tok==='BS'){
          calcState.expr = calcState.expr.slice(0,-1);
          const tryRes = evaluateExpression(calcState.expr || '0');
          if(tryRes===null) calcState.value='0'; else setValueFromNumber(tryRes);
          updateCalcUI();
          return;
        }
        if(tok==='='){
          const expr = (calcState.expr || '0').trim();
          const res = evaluateExpression(expr);
          if(res===null){ toast('Expressão inválida.'); return; }
          const pretty = expr.replace(/\s+/g,' ');
          calcState.history.unshift({expr: pretty, val: String(res)});
          calcState.history = calcState.history.slice(0,20);
          setValueFromNumber(res);
          calcState.expr='';
          persistCalc();
          updateCalcUI();
          return;
        }
        if(tok==='MC'){ calcState.mem=0; persistCalc(); updateCalcUI(); return; }
        if(tok==='MR'){ calcState.expr += String(calcState.mem); updateCalcUI(); return; }
        if(tok==='M+'){ const n=safeNumber(calcState.value); if(n===null) return; calcState.mem += n; persistCalc(); updateCalcUI(); return; }
        if(tok==='M-'){ const n=safeNumber(calcState.value); if(n===null) return; calcState.mem -= n; persistCalc(); updateCalcUI(); return; }

        if(['+','-','*','/'].includes(tok)){
          calcState.expr = calcState.expr.replace(/[+\-*/]\s*$/,'');
          calcState.expr += normalizeOperator(tok);
          updateCalcUI();
          return;
        }

        appendToken(tok);
        const res = evaluateExpression(calcState.expr);
        if(res!==null) setValueFromNumber(res);
        updateCalcUI();
      }

      // --------- Checklist & Lembretes (faltava implementação) ---------
      const quickCheckEl = $('#quickCheck');
      const checkNewEl = $('#checkNew');
      const btnCheckAdd = $('#btnCheckAdd');
      const remindersEl = $('#reminders');
      const btnRemClear = $('#btnRemClear');
      const btnRemUrgent = $('#btnRemUrgent');

      const quickCheckKey = 'bb_quick_checks';
      const remindersKey = 'bb_reminders';

      const quickChecksState = { items: [] };
      const remindersState = { items: [] };

      function persistQuickChecks(){
        try{ localStorage.setItem(quickCheckKey, JSON.stringify(quickChecksState.items)); }catch{}
      }
      function loadQuickChecks(){
        try{
          const raw = localStorage.getItem(quickCheckKey);
          if(raw) quickChecksState.items = JSON.parse(raw);
        }catch{}
      }

      function persistReminders(){
        try{ localStorage.setItem(remindersKey, JSON.stringify(remindersState.items)); }catch{}
      }
      function loadReminders(){
        try{
          const raw = localStorage.getItem(remindersKey);
          if(raw) remindersState.items = JSON.parse(raw);
        }catch{}
      }

      function toastOnce(type){
        if(type==='urgent') toast('Lembrete urgente!');
        if(type==='pomodoro') toast('Pomodoro (25m) iniciado!');
        if(type==='break') toast('Pausa (5m) iniciada!');
      }

      function renderQuickChecks(){
        if(!quickCheckEl) return;
        quickCheckEl.innerHTML='';
        const items = quickChecksState.items;
        if(items.length===0){
          const empty=document.createElement('div');
          empty.style.color='var(--muted)';
          empty.style.fontSize='12px';
          empty.textContent='Nenhum item ainda. Adicione acima.';
          quickCheckEl.appendChild(empty);
          return;
        }
        items.forEach((it, idx)=>{
          const row=document.createElement('div');
          row.style.display='flex';
          row.style.alignItems='center';
          row.style.justifyContent='space-between';
          row.style.gap='10px';
          row.style.padding='10px';
          row.style.border='1px solid rgba(255,255,255,.12)';
          row.style.borderRadius='14px';
          row.style.background='rgba(255,255,255,.05)';

          const left=document.createElement('div');
          left.style.display='flex';
          left.style.alignItems='center';
          left.style.gap='10px';
          left.style.minWidth='0';

          const cb=document.createElement('input');
          cb.type='checkbox';
          cb.checked=!!it.done;
          cb.style.width='18px';
          cb.style.height='18px';

          const txt=document.createElement('div');
          txt.style.fontFamily='var(--sans)';
          txt.style.color='var(--text)';
          txt.style.fontSize='12px';
          txt.style.whiteSpace='nowrap';
          txt.style.overflow='hidden';
          txt.style.textOverflow='ellipsis';
          txt.textContent=it.text;
          if(it.done) txt.style.textDecoration='line-through';

          cb.addEventListener('change', ()=>{
            quickChecksState.items[idx].done = cb.checked;
            persistQuickChecks();
            renderQuickChecks();
          });

          left.appendChild(cb);
          left.appendChild(txt);

          const del=document.createElement('button');
          del.className='smallbtn';
          del.textContent='Remover';
          del.style.padding='8px 10px';
          del.addEventListener('click', ()=>{
            if(!confirm('Remover este item?')) return;
            quickChecksState.items.splice(idx,1);
            persistQuickChecks();
            renderQuickChecks();
          });

          row.appendChild(left);
          row.appendChild(del);
          quickCheckEl.appendChild(row);
        });
      }

      function renderReminders(){
        if(!remindersEl) return;
        remindersEl.innerHTML='';
        const items = remindersState.items;
        if(items.length===0){
          const empty=document.createElement('div');
          empty.style.color='var(--muted)';
          empty.style.fontSize='12px';
          empty.textContent='Sem lembretes.';
          remindersEl.appendChild(empty);
          return;
        }
        items.slice().reverse().forEach((it, idx)=>{
          const wrapper=document.createElement('div');
          wrapper.style.display='flex';
          wrapper.style.justifyContent='space-between';
          wrapper.style.alignItems='center';
          wrapper.style.gap='10px';
          wrapper.style.border='1px solid rgba(255,255,255,.12)';
          wrapper.style.borderRadius='14px';
          wrapper.style.background='rgba(255,255,255,.05)';
          wrapper.style.padding='10px';
          wrapper.style.marginBottom='10px';

          const left=document.createElement('div');
          left.style.minWidth='0';
          const title=document.createElement('div');
          title.style.fontWeight='850';
          title.style.fontSize='12px';
          title.textContent=it.label;
          const time=document.createElement('div');
          time.style.color='var(--muted)';
          time.style.fontSize='11px';
          time.textContent=new Date(it.createdAt).toLocaleTimeString('pt-BR', {hour:'2-digit', minute:'2-digit'});
          left.appendChild(title);
          left.appendChild(time);

          const del=document.createElement('button');
          del.className='smallbtn';
          del.textContent='Fechar';
          del.style.padding='8px 10px';
          del.addEventListener('click', ()=>{
            const realIndex = remindersState.items.length - 1 - idx;
            remindersState.items.splice(realIndex,1);
            persistReminders();
            renderReminders();
          });

          wrapper.appendChild(left);
          wrapper.appendChild(del);
          remindersEl.appendChild(wrapper);
        });
      }

      function addQuickCheck(text){
        quickChecksState.items.unshift({ text, done:false, createdAt: Date.now() });
        quickChecksState.items = quickChecksState.items.slice(0,20);
        persistQuickChecks();
        renderQuickChecks();
      }

      function addReminder(label){
        remindersState.items.push({ label, createdAt: Date.now() });
        remindersState.items = remindersState.items.slice(-20);
        persistReminders();
        renderReminders();
      }

      btnCheckAdd?.addEventListener('click', ()=>{
        const v = (checkNewEl?.value||'').trim();
        if(!v) return;
        addQuickCheck(v);
        if(checkNewEl) checkNewEl.value='';
        toast('Item adicionado.');
      });
      checkNewEl?.addEventListener('keydown', (e)=>{
        if(e.key==='Enter'){
          btnCheckAdd?.click();
          e.preventDefault();
        }
      });

      btnRemClear?.addEventListener('click', ()=>{
        if(!confirm('Limpar todos os lembretes?')) return;
        remindersState.items=[];
        persistReminders();
        renderReminders();
        toast('Lembretes limpos.');
      });

      btnRemUrgent?.addEventListener('click', ()=>{
        addReminder('Urgente agora');
        toastOnce('urgent');
      });

      // Botões pomodoro/break (apenas cria lembrete + opcional visual)
      $$('button[data-rem="pomodoro"]').forEach(btn=>{
        btn.addEventListener('click', ()=>{
          addReminder('Pomodoro (25m)');
          toastOnce('pomodoro');
        });
      });
      $$('button[data-rem="break"]').forEach(btn=>{
        btn.addEventListener('click', ()=>{
          addReminder('Pausa (5m)');
          toastOnce('break');
        });
      });

      loadQuickChecks();
      loadReminders();
      renderQuickChecks();
      renderReminders();
    </script>

  </body>
</html>
